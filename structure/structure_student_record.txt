#include <stdio.h>
 struct student
 {
 int student_id;
 char name[10]; 
 char grade;
 float marks[5];
 float average_marks;
};
 int main() {
 struct student one[3]; 
 int i,j;
 for(i=0;i<3;i++)
 {
 printf("enter the student id:"); 
 scanf("%d",&one[i].student_id);
 printf("enter the name:"); 
 scanf("%s",one[i].name);
 printf("enter the marks:");
 int sum=0;
 for(j=0;j<5;j++){
 scanf("%f",&one[i].marks[j]);
 sum +=one[i].marks[j];
}
one[i].average_marks = sum/5.0; 
printf("id: %d \n ",one[i].student_id); 
printf("name: %s \n",one[i].name); 
for(j=0;j<5;j++){
printf("marks:%.2f \n",one[i].marks[j]);
}   
printf("average_marks: %.2f \n", one[i].average_marks); 
if (one[i].average_marks >90.0 && one[i].average_marks <100.0)
{
 printf("A+ \n");
} 
else if(one[i].average_marks >80.0&&one[i].average_marks <90.0)
{
 printf("A \n");
}
 else if (one[i].average_marks >70.0&&one[i].average_marks <80.0) 
 {
 printf("B+ \n");
 }
 else if (one[i].average_marks >60.0&&one[i].average_marks <70.0)
 {
 printf("B \n");
 }
 else {
 printf("FAIL \n");
}
}
return 0;
}
