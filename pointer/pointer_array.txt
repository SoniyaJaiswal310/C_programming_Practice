#include<stdio.h>
int main()
{
int n,i;
printf("enter the number of elements of the array: ");
scanf("%d", &n);
int arr[n];
for(i=0;i<n;i++){
printf("enter the elements of the array: ");
scanf("%d", &arr[i]);
}
int *b = arr;
printf("value of 1st element:%d\n",*b);
printf("address of the 1st element:%p \n",b);
for(i=0;i<n;i++){
    printf("value: %d \t", b[i]);
    printf("address: %p \n", b+i);
}
return 0;
}
