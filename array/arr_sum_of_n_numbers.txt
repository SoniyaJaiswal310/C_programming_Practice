#include<stdio.h>
int main()
{
    int n,i,sum =0;
    printf("enter the number for elements in the array: ");
    scanf("%d", &n);
    int arr[n];
    for(i=0;i<n;i++)
    {
        printf("enter the elements in the array: ");
        scanf("%d", &arr[i]);
        sum+=i;
    }
    printf("sum = %d", sum);
    return 0;
}
